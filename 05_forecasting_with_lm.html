ts <- ts1 |>
dplyr::select(period, y = value) |>
dplyr::filter(period > 1986) |>
as_tsibble(index = period)  

plot_ly(data = ts, x = ~period, y = ~y, type = "scatter", mode = "lines") 
ts_stl_d <- ts |>
  model(
    STL(y ~ trend(window = 11) +
                   season(window = 7),
    robust = TRUE),
  ) |>
  components()
plot_decomposition(ts_stl_d, var = "y", outliers = TRUE)
